        function pushLog(message, actor) {

        function updateStats() {
            const pending = db.applicants.filter(a => a.status === 'Pending').length;
            const approved = db.applicants.filter(a => a.status === 'Approved').length;
            const rejected = db.applicants.filter(a => a.status === 'Rejected').length;
            let totalWarnings = 0; const approvedNames = db.applicants.filter(a => a.status === 'Approved').map(a => a.discordName);
            approvedNames.forEach(dName => { totalWarnings += (db.warnings[dName] || 0); });
            document.getElementById('stat-pending').innerText = pending;
            document.getElementById('stat-approved').innerText = approved;
            document.getElementById('stat-rejected').innerText = rejected;
            document.getElementById('stat-warnings').innerText = totalWarnings;
        }

        function updateStatus(id, newStatus) {
            const applicant = db.applicants.find(a => a.id === id);
            if(applicant) {
                applicant.status = newStatus; pushLog(`Applicant "${applicant.name}" status changed to ${newStatus}.`, session.managementRole);
                saveDatabaseToStorage(); updateStats(); renderApplicants(); renderStaffTable(); showToast(`Status updated successfully.`, "success");
            }
        }

        function deleteApplication(id) {
            const app = db.applicants.find(a => a.id === id);
            if(app) {
                if(!confirm(`Are you sure you want to purge ${app.name}'s file?`)) return;
                pushLog(`Permanently deleted recruitment sheet index of "${app.name}".`, session.managementRole);
                db.applicants = db.applicants.filter(a => a.id !== id);
                saveDatabaseToStorage(); updateStats(); renderApplicants(); renderStaffTable(); showToast("Application sheet purged.", "info");
            }
        }

        function addWarning(dName) {
            if(session.managementRole !== 'Head Operator') { showToast("Access Denied: assignment requires Head Operator permissions.", "error"); return; }
            db.warnings[dName] = (db.warnings[dName] || 0) + 1;
            pushLog(`Warning infraction logged against user "${dName}".`, 'Head Operator');
            if(db.warnings[dName] > 3) {
                pushLog("Auto-Termination Checked: Warning limit exceeded.", 'SYSTEM');
                const staff = db.applicants.find(a => a.discordName === dName); if(staff) staff.status = 'Rejected';
                showToast(`Personnel entry ${dName} auto-removed.`, "error");
            } else { showToast(`Strike index recorded against ${dName}.`, "warning"); }
            saveDatabaseToStorage(); updateStats(); renderStaffTable();
        }

        function terminateEmployee(id) {
            if(session.managementRole !== 'Head Operator') { showToast("Access Denied: Requires Head Operator permissions.", "error"); return; }
            const staff = db.applicants.find(a => a.id === id);
            if(staff) {
                staff.status = 'Rejected'; pushLog(`Personnel clearance roster for ${staff.name} explicitly revoked.`, 'Head Operator');
                saveDatabaseToStorage(); updateStats(); renderApplicants(); renderStaffTable(); showToast("Staff deployment asset terminated.", "info");
            }
        }

        function updateEmployeeName(id, currentElement) {
            if (session.managementRole === 'Operator') {
                showToast("Access Denied: Operators cannot change names.", "error");
                routeCurrentSessionState(); return;
            }
            const staff = db.applicants.find(a => a.id === id); const cleanName = currentElement.value.trim();
            if(!cleanName) { currentElement.value = staff.discordName; return; }
            if(staff && staff.discordName !== cleanName) {
                const oldName = staff.discordName;
                if(db.warnings[oldName]) { db.warnings[cleanName] = db.warnings[oldName]; delete db.warnings[oldName]; }
                staff.discordName = cleanName; pushLog(`Personnel pointer updated from "${oldName}" to "${cleanName}".`, session.managementRole);
                saveDatabaseToStorage(); updateStats(); showToast("Roster handle index corrected.", "success");
            }
        }
